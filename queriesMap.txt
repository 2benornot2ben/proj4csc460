ADMINq2: SELECT u.username AS username, u.email AS email, SUM(i.amount) AS total_amount_owed, MAX(c.created_at) AS last_conversation_date FROM instuser u JOIN invoice i ON u.userid = i.userid LEFT JOIN conversation c ON u.userid = c.userid WHERE UPPER(i.payment_status) = 'UNPAID' GROUP BY u.username, u.email ORDER BY total_amount_owed DESC, u.username
